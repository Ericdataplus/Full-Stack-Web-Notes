Install the React one - https://github.com/tailwindlabs/heroicons#vue
- npm install @heroicons/react

Then use like this

```jsx
import { BeakerIcon } from '@heroicons/react/solid'

function MyComponent() {
  return (
    <div>
      <BeakerIcon className="h-5 w-5 text-blue-500"/>
      <p>...</p>
    </div>
  )
}
```
